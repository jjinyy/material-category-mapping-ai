# 반드시 존재해야 할 파일
trained_model/ (학습 완료 후 자동 생성됨)
faiss_index.bin와 faiss_mapping.json
이 두 개는 build_faiss_index.py 실행으로 생성됨

# 실행순서 (최초 1회)
# 1. 모델 학습
python -m train.init_train
# 2. FAISS 인덱스 구축
python -m train.build_faiss_index
# 3. UI 실행 (Main Launcher)
python -m scripts.main

cjCategoryMapping/
│
├── train/                   # 모델 학습 관련 코드
│   ├── __init__.py           # 패키지 인식용 (비어있어도 반드시 필요)

│   ├── train_model.py        # 모델 학습 스크립트 / 피드백으로 모델 재학습 (fine-tuning)
│   ├── generate_training_data.py  # 훈련 데이터 생성 스크립트 / 사용자 피드백 기반 학습 데이터 생성
│   ├── init_train.py         # 학습 초기화 스크립트 (옵션 세팅 등) / 최초 Pretrained 모델로 기본 모델 저장
│
├── ui/                       # 사용자 인터페이스(UI) 관련 코드
│   ├── __init__.py            # 패키지 인식용
│   ├── ui.py                  # 사용자 입력 및 결과 출력 인터페이스 / GUI 프로그램 — 사용자 입력 받고 결과 보여줌, 재학습까지 클릭으로 가능
│
├── data/                      # 데이터 파일 저장 폴더

├── models/                     # 모델, 인덱스, 체크포인트 파일
│   ├── __init__.py              # 패키지 인식용

│   ├── checkpoints/             # 학습 중간 체크포인트 저장 폴더
│   ├── faiss_index.bin          # FAISS 인덱스 파일 (검색용 벡터 저장)

│   ├── __init__.py               # 패키지 인식용
│   ├── main.py                   # 메인 실행 파일 (UI 호출, 모델 로딩 등) / 모델 + 인덱스 로드 → 자재명 분류 기능
